pwd	#показать путь до текущего каталога

ls	#показать список файлов в текущем каталоге
	-a	#показать скрытые файлы
	-l	#показать подробную информацию о файлах
	-h(+l)	#показать удобный размер файла

cd	#перемещение по каталогам. перейти в домашний каталог
	<путь>	#перейти в заданный каталог
	..	#перейти на уровень вверх

touch <имя файла>	#создать файл

mkdir <имя каталога>	#создать каталог

echo <текст>	#вывод текста
	*	#любое количество любых символов
	?	#один символ. echo *.? выведет все файлы с расшитением из одного символа
	[]	#любой символ из множества
	^	#исключить символ. echo [^_]* выведет все файлы кроме начинающихся с _
	{}	#любая цепочка символов из перечисленных в скобках. echo *.{jpg,png,gif} выведет файлы только с указанными расширениями

regexr.com	#тренажер регулярных выражений

rm <имя файла>	#удалить файл или пустой каталог
	-r	#удалить все файлы в каталоге и сам каталог
	-f	#игнорировать любые ошибки
	-i	#спрашивать разрешение перед каждым удалением

ps	#показать выполняющиеся процессы
	ax	#показать системные процессы
	u	#показать дополнительную информацию

процессы
	CTRL+Z	#переводит процесс в фон
	CTRL+D	#имитирует конец файла
	CTRL+C	#завершить процесс, если она ничего не считывает
	CTRL+\	#принудительно завершить процесс
	CTRL+S	#поставить терминал на паузу
	CTRL+Q	#отключить паузу
	kill PID	#убить процесс
		-9 PID		#\
		-KILL %1	#-убить процесс принудительно
			PID	#номер процесса
			%1	#номер фоновой задачи
	&	#запустить процесс в фоновом режиме
	jobs	#показать фоновые процессы, запущенные пользователем
	bg %1	#продолжить выполнение задачи в фоне
	fg %1	#сделать фоновый процесс текущим

cat <имя файла>	#показать содержимое файла
	-n	#нумерует строки
	-b	#нумерует непустые строки
	-s	#сжимает несколько пустых строк, идущих подряд
	-t	#печатает невидимые символы и превращает табы в ^I
	-e	#печатает невидимые символы и добавляет в коней каждой строки $

	<	#считать из файла. cmd1 < file1 запустить программу cmd1 и передать в нее содержимое file1
	>	#перезаписать в файл. cat file1 > file2 создаст file2 и запишет в него file1 или заменит содержимое file2 на file1
	>>	#записать в конец файла. cat file1 >> file2 содержимое file1 записать в конец file2
	>&	#объединить 2 потока и вывести в файл
	|	#конвейер - запускает 2 программы и направляет вывод одной программы на ввод другой или записывает в файл
		cmd1 2>&1 | cmd2	cmd1 2>&1 | file1
потоки
	0	#стандартного ввода
	1	#стандартного вывода
	2	#для вывода сообщений об ошибках

grep	#поиск подстроки в строке. cat hello -n | grep ".[n]"
	-e	#флаг паттерна, после него обязательно идет строка с паттерном
	-F	#интерпретировать паттерны, как строки
	-f	#достать паттерны из файла, после этого флага идет название файла
	-i	#игнорировать регистр
	-v	#инвертировать результат
	-o	#выводить только совпавшую часть строки
	-c	#выводить только количество совпавших строк

less	#покрутка и поиск по результату вывода программы. cat 123.txt | less

history		#выводит историю введенных команд
	!1	#выполнит первую команду из history

	!!	#выполнит последнюю команду
	!gcc:2	#вытащит второе слово из команды
